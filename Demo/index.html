jquery.moduleCraft

使用不同的模块，产生不同的用法

使用方法

在使用前先引入模块文件，入参"data"的具体数据因模块而异

初始化:

                  $("#dome").moduleCraft({
    //使用的模块
    module: "bootstrapLabel",
    //生成html的输入数据
    "data": {data:
      [
        {"key": "a", "name": "a", "type": "default"},
        {"key": "b", "name": "b", "type": "primary"},
        {"key": "c", "name": "c", "type": "success"},
        {"key": "d", "name": "d ", "type": "info"},
        {"key": "e", "name": "e", "type": "warning"},
        {"key": "f", "name": "f", "type": "danger"}
      ]
    },
    //回调
    "callback": function (e, key, val) {
        console.log(e, key, val);
    }
});

触发一个元素

                  $("#dome").moduleCraft("trigger",key)

切换禁用

                  $("#dome").moduleCraft("disable"[,true/false])
